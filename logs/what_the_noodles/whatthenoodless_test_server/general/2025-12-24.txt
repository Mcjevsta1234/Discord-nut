**`client/src/components/Console.jsx`** ‚Äì mock terminal UI  

```jsx
import { useState } from "react";

export default function Console() {
  const [log, setLog] = useState([]);
  const [cmd, setCmd] = useState("");

  const run = async () => {
    const res = await fetch("/api/command", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ command: cmd }),
    });
    const { output } = await res.json();
    setLog([...log, `> ${cmd}`, output]);
    setCmd("");
  };

  return (
    <div className="bg-gray-900 text-green-400 p-4 rounded">
      {log.map((l, i) => (
        <div key={i}>{l}</div>
      ))}
      <input
        className="w-full bg-transparent outline-none"
        value={cmd}
        onChange={(e) => setCmd(e.target.value)}
        onKeyDown={(e) => e.key === "Enter" && run()}
        placeholder="type a command..."
      />
    </div>
  );
}
```

**`server/mcCommands.js`** ‚Äì minimal command parser  

```js
const exec = require("child_process").exec;

const handlers = {
  status: () => "üü¢ Server is online",
  stop: () => {

module.exports = (cmd) => {
  const key = cmd.trim().replace("/", "").split(" ")[0];
  return handlers[key] ? handlers[key]() : `‚ùì Unknown command: ${cmd}`;
};
```

**`server/index.js`** ‚Äì API endpoint  

```js
const express = require("express");
const handleCmd = require("./mcCommands");
const app = express();

app.use(express.json());
app.post("/api/command", (req, res) => {
  const { command } = req.body;
  const output = handleCmd(command);
  res.json({ output });
});

app.listen(4000, () => console.log("API running on :4000"));
```
